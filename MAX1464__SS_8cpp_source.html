69 uint16_t MAX1464_SS::wordShiftIn() const
70 {
71  if(_3wireMode) {
72  writeNibble(IMR_3WIRE, IRSA_IMR);
73  pinMode(_spi_datain, INPUT);
74  }
75  uint16_t w = 0;
76  digitalWrite(_spi_clock, LOW);
77  digitalWrite(_chipSelect, LOW);
78  w |= (shiftIn(_spi_datain, _spi_clock, MSBFIRST) << 8);
79  w |= (shiftIn(_spi_datain, _spi_clock, MSBFIRST));
80  digitalWrite(_chipSelect, HIGH);
81  if(_3wireMode) {
82  pinMode(_spi_dataout, OUTPUT);
83  digitalWrite(_spi_dataout, LOW);
84  }
85  return w;
86 }
