250 boolean AbstractMAX1464::writeHexLineToFlashMemory(const String hexline)
251 {
252  if(hexline.charAt(0) != ':')
253  return false;
254  unsigned int i = 1;
255  uint8_t byteCount = strtoul(hexline.substring(i,i+2).c_str(),NULL,16);
256  i+=2;
257  uint16_t address = strtoul(hexline.substring(i,i+4).c_str(),NULL,16);
258  i+=4;
259  uint8_t recordType = strtoul(hexline.substring(i,i+2).c_str(),NULL,16);
260  i+=2;
261  byte data[byteCount];
262  uint16_t sum = byteCount+(address >> 8)+(address & 0xff)+recordType;
263  for (uint8_t count = 0; count < byteCount; ++count) {
264  byte b = strtoul(hexline.substring(i,i+2).c_str(),NULL,16);
265  sum += b;
266  data[count] = b;
267  i+=2;
268  }
269  sum += strtoul(hexline.substring(i,i+2).c_str(),NULL,16); //checksum
270  if((sum & 0xff) != 0) {
271  Serial.print("Wrong checksum ");
272  printHex16(sum);
273  return false;
274  }
275  if(recordType == 0x01) { //end of file
276  EOFReached = true;
277  return true;
278  }
279  else
280  EOFReached = false;
281  if(recordType != 0x0) {
282  Serial.println("Wrong recordType");
283  return false;
284  }
285  uint16_t addr = address;
286  for (uint8_t count = 0; count < byteCount; ++count)
287  writeByteToFlash(data[count],addr++);
288  return true;
289 }

300 void AbstractMAX1464::readFlashPartition(
301  const FLASH_PARTITION partition) const {
302  haltCpu();
303  uint8_t temp[16];
304  uint8_t i = 0;
305  uint16_t partition_size = 0x1000;
306  if(partition == 1) {
307  writeCR(CR_SELECT_FLASH_PARTITION_1);
308  partition_size = 0x80;
309  }
310  for(uint16_t addr = 0; addr < partition_size; addr++) {
311  // set address
312  setFlashAddress(addr);
313  copyFlashToDhr();
314 
315  temp[i++] = wordShiftIn() & 0xff;
316 
317  if(i == 16) {
318  uint8_t checksum = 0;
319  Serial.print(":10"); // byte count
320  checksum += 0x10;
321  union { uint16_t val; struct { uint8_t lsb; uint8_t msb; }; } a;
322  a.val = addr - 15;
323  if(a.msb < 0x10)
324  Serial.print("0");
325  Serial.print(a.msb, HEX);
326  if(a.lsb < 0x10)
327  Serial.print("0");
328  Serial.print(a.lsb, HEX);
329  checksum += a.msb;
330  checksum += a.lsb;
331  Serial.print("00"); // record type
332  for(uint8_t j = 0; j<16; j++) {
333  uint8_t b = temp[j];
334  if(b <= 0xf)
335  Serial.print("0");
336  Serial.print(b, HEX);
337  checksum += b;
338  }
339  checksum = ~checksum + 1;
340  if(checksum < 0x10)
341  Serial.print("0");
342  Serial.println(checksum, HEX);
343  i = 0;
344  }
345  }
346  Serial.println(":00000001FF");
347 }
